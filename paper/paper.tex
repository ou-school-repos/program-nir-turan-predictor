\begin{abstract}
We investigate the minimization of graph homomorphisms among trees
into fixed target graphs $H$, addressing open problems from
Galvin et al.~\cite{galvin2025,galvin2026}. We computationally verify the odd-path
minimizer theorem for $H = P_3, P_5, P_7, P_9$ up to source trees on
$n = 22$ vertices, establishing $P_5$ as the transition point from
total spectral degeneracy to strict unique minimality. We push the
lower bound on the smallest simple Leontovich graph from $m \ge 4$ to
$m \ge 12$ via exhaustive enumeration of all 1,018,690,325 connected simple
graphs on $m \le 11$ vertices. We prove a \emph{1-Positive Eigenvalue Rule}
showing that 2-orbit and 3-orbit spherically symmetric trees can never
be Leontovich, and we analytically pinpoint the exact crossover
threshold $n = 13$ for the known Leontovich graph $T(7,1,9)$, proving
via monotonicity of the spectral ratio that this threshold is
non-oscillatory. Via simulated annealing, we discover a
76-vertex Leontovich graph $T(7,1,[9^5,8^2])$---the smallest
known---and show that its Leontovich property is optimally tuned by the
active secondary eigenvalue $\lambda_2^* \approx 3.0380$ of its asymmetric quotient matrix.
\end{abstract}

\section{Introduction}

For graphs $G$ and $H$, a \emph{graph homomorphism} $\varphi \colon
V(G) \to V(H)$ is a map preserving adjacency. The number of such maps,
$\Hom(G, H)$, encodes fundamental combinatorial information.
Sidorenko~\cite{sidorenko1991} established that stars maximize
$\Hom(T, H)$ among trees $T$ on $n$ vertices for any target $H$.

The dual question---which trees \emph{minimize} homomorphisms---remains
largely open. Galvin and Tetali~\cite{galvin-tetali} conjectured that
the path $P_n$ minimizes $\Hom(T, H)$ among $n$-vertex trees for all
$H$. Leontovich~\cite{leontovich1989} first observed that this fails
for certain targets;
Csikv\'ari and Lin~\cite{csikvari-lin} systematically developed the
theory via the KC-transformation. Recently, Galvin, McMillon, Redlich,
and Nir~\cite{galvin2025} demonstrated explicit bipartite counterexamples,
coining \emph{Leontovich graphs}: targets $H$ for which $P_n$ does not
minimize $\Hom(T, H)$ among trees. Subsequent work by Galvin et al.~\cite{galvin2026}
introduced the dual concept of \emph{Hoffman-London graphs} to classify
targets where paths strictly minimize.

\subsection{Summary of contributions}
\begin{enumerate}[label=(\roman*)]
  \item Computational verification of the odd-path minimizer theorem
    for $H = P_3, P_5, P_7, P_9$ up to $n = 22$, revealing a sharp
    transition: $P_3$ is totally degenerate (depending only on the
    bipartition, cf.\ Theorem~2.6 of \cite{galvin2026}), $P_5$ minimizes
    but \emph{not uniquely} (exponentially growing structural equivalence classes from
    2-shift invariance), and $P_{\ge 7}$ exhibits strict unique minimality. This directly
    addresses Problem~6.3 of \cite{galvin2026}
    (Section~\ref{sec:odd-path}).
  \item A \emph{1-Positive Eigenvalue Rule} (Theorem~\ref{thm:spectral})
    proving that 2-orbit and 3-orbit trees cannot be Leontovich.
  \item An exact analytical determination that $n = 13$ is the
    non-oscillatory crossover threshold for $T(7,1,9)$
    (Theorem~\ref{thm:threshold}).
  \item Pushing the lower bound on the smallest simple Leontovich graph from
    $m \ge 4$ to $m \ge 12$ (via exhaustive enumeration of all
    1,018,690,325 connected simple graphs on $m \le 11$ vertices). This
    resolves Problem~4.3 of \cite{galvin2025} for simple graphs.
  \item Exhaustive search of all 5,357 spherically symmetric trees with
    $|V| < 78$, ruling out smaller Leontovich graphs in this family.
  \item Algebraic proof that $n = 5$ crossovers are impossible for all
    $T(x,y,z)$ (Proposition~\ref{prop:n5}), and a threshold landscape
    showing that $T(7,1,9)$ is the global vertex-count minimum.
\end{enumerate}

\section{Preliminaries}
\label{sec:prelim}

\begin{definition}[$E_n$]
  The \emph{near-path tree} $E_n$ is the tree on $n$ vertices obtained
  from the path $P_{n-1}$ by attaching a pendant vertex to the
  third-from-last vertex.
\end{definition}

\begin{definition}[$T(x,y,z)$]
  For positive integers $x, y, z$, the \emph{spherically symmetric tree}
  $T(x,y,z)$ is a rooted tree with 4 orbits: a root (orbit~0, 1~vertex),
  $x$ children of the root (orbit~1), $xy$ grandchildren (orbit~2),
  and $xyz$ leaves (orbit~3). Total vertices: $|V| = 1 + x + xy + xyz$.
\end{definition}

\begin{definition}[$E_n^{(d)}$]
  The \emph{near-path family} $E_n^{(d)}$ consists of trees on $n$
  vertices obtained from $P_{n-1}$ by attaching a pendant vertex at
  depth $d$ from the end. When $d = 2$, this recovers $E_n$.
\end{definition}

\section{The Odd-Path Minimizer Theorem}
\label{sec:odd-path}

\begin{theorem}[Odd-Path Minimizer, Csikv\'ari and Lin \cite{csikvari2015}]
  For every odd path $P_k$ and every $n$-vertex tree $T$,
  $\Hom(P_n, P_k) \le \Hom(T, P_k)$.
\end{theorem}

\subsection{Computational verification}

We exhaustively explored this theorem computationally for
$H \in \{P_3, P_5, P_7, P_9\}$ and all $n$-vertex trees with
$n \le 22$. Tree enumeration uses the Beyer--Hedetniemi canonical
level-sequence algorithm implemented in C++, producing all
non-isomorphic trees. Tree counts are cross-validated against
OEIS A000055~\cite{oeis_A000055}.

\subsection{The hierarchy of rigidity}

The spectral structure of $P_k$ governs the degree of uniqueness with
which $P_n$ minimizes homomorphisms. We identify three regimes.

\subsubsection{\texorpdfstring{$P_3$}{P3}: Total degeneracy}

\begin{proposition}
  \label{prop:p3}
  For any tree $T$ with bipartition $(A, B)$,
  \[
    \Hom(T, P_3) = 2^{|A|} + 2^{|B|}.
  \]
  In particular, $\Hom(P_n, P_3) = 2^{\lceil n/2 \rceil}
  + 2^{\lfloor n/2 \rfloor}$.
\end{proposition}

Since $P_n$ has the most balanced bipartition among $n$-vertex trees,
it achieves the minimum, but any tree sharing the same bipartition sizes
ties it exactly. The path wins, but not uniquely. This is a special case
of Theorem~2.6 of~\cite{galvin2026}, since $P_3 \cong K_{1,2}$.

\subsubsection{\texorpdfstring{$P_5$}{P5}: Interior degeneracy and 2-shift invariance}

The adjacency matrix $A = A_{P_5}$ satisfies
\begin{equation}
  \label{eq:shift}
  A^3 \mathbf{1} = 3\, A \mathbf{1},
\end{equation}
where $\mathbf{1}$ is the all-ones vector. This identity implies that
the cube of the transfer matrix acts as a scalar multiple of the
matrix itself on the constant vector. Combinatorially, moving a pendant
edge two positions along the interior of a path preserves the
homomorphism count into $P_5$: length-3 walks from vertex $v$ to the
all-ones vector are exactly $3\times$ the length-1 walks, so pendant
relocations that shift by 2 steps produce identical walk sums.
This generates exponentially growing structural equivalence classes ($\sim\!200$ ties at $n = 20$).

The eigenvalues of $P_5$ are $\pm\sqrt{3}, \pm 1, 0$, so
$\lambda_1^2 = 3$ exactly, and the ratio
$\Hom(P_{n+2}, P_5) / \Hom(P_n, P_5) = 3$ for all odd $n$.

\subsubsection{\texorpdfstring{$P_{\ge 7}$}{P>=7}: Strict unique minimality}

For $k \ge 7$, the eigenvalues of $A_{P_k}$ are
$2\cos\bigl(\frac{j\pi}{k+1}\bigr)$ for $j = 1, \ldots, k$.
The squared spectral radius is $\lambda_1^2 = 4\cos^2\bigl(\frac{\pi}{k+1}\bigr) = 2 + 2\cos\bigl(\frac{2\pi}{k+1}\bigr)$.
By Niven's Theorem, $\cos(q\pi)$ is rational for rational $q$ if and only if
$q \in \{0, \pm 1/3, \pm 1/2, \pm 2/3, \pm 1\}$. Thus, $\lambda_1^2 \in \mathbb{Q}$ if and only if $k \in \{1, 2, 3, 5\}$.

For all $k \ge 6$, the squared dominant eigenvalue $\lambda_1^2$ is strictly irrational.
This permanent irrationality breaks any potential integer shift-symmetry of the form
$A^m \mathbf{1} = cA \mathbf{1}$ (which would require $\lambda_1^m \in \mathbb{Q}$ or $\lambda_1^{m-1} \in \mathbb{Q}$), isolating $P_n$ as the strict unique minimizer.

\begin{remark}[Irrationality]
  The transition to strict unique minimality occurs exactly at $P_7$ because it
  is the first odd path whose squared spectral radius ($2+\sqrt{2}$)
  is irrational. While still constructible in real radicals, its irrationality
  prevents the formation of the exact degeneracy classes seen in $P_5$.
\end{remark}

\section{Analytical Properties of Leontovich Graphs}
\label{sec:leontovich}

\subsection{The Spectral Crossover Theorem}

\begin{theorem}[1-Positive Eigenvalue Rule]
  \label{thm:spectral}
  Let $H$ be a rooted bipartite graph whose automorphism orbits respect
  the bipartition (i.e., no orbit contains vertices from both sides).
  If the quotient matrix of $H$ has exactly one positive eigenvalue,
  then $H$ cannot be Leontovich.
\end{theorem}

\begin{proof}
  Since the orbits respect the bipartition, the quotient matrix is
  itself bipartite, so its spectrum is symmetric about zero.
  Because the base vector $\mathbf{1}$ and the initial degree vector
  are constant on automorphism orbits, the entire dynamic programming
  sequence $\{A^k \mathbf{1}\}$ remains in the column space of the
  quotient matrix, so the full adjacency spectrum is unnecessary.
  If there is exactly one positive eigenvalue $\lambda_1$, the non-zero
  active spectrum is $\{\lambda_1, -\lambda_1\}$. By the transfer matrix
  decomposition over this quotient space,
  \[
    \Hom(G_n, H) = c\,\lambda_1^n + d\,(-\lambda_1)^n.
  \]
  For odd $n$, this reduces to $(c - d)\,\lambda_1^n$. There are no
  secondary exponential terms: the ratio
  $\Hom(E_n, H) / \Hom(P_n, H)$ is a strict constant for all odd~$n$.
  Since $P_n$ minimizes at small $n$ (verified computationally), it is
  locked in for all $n$.
\end{proof}

\begin{corollary}
  \label{cor:2_and_3_orbit_trees}
  All 2-orbit trees (stars $S_m$) and 3-orbit trees $T(x,y)$ have
  exactly one positive eigenvalue. Hence no star or 3-orbit tree is
  Leontovich.
\end{corollary}

\subsection{Four-orbit trees: the unconditional discriminant}

For 4-orbit trees $T(x,y,z)$, the characteristic polynomial of the
quotient matrix is
\begin{equation}
  \label{eq:charpoly}
  \lambda^4 - (x+y+z)\,\lambda^2 + xz = 0.
\end{equation}
Setting $s = x + y + z$, the discriminant of the quadratic in
$\lambda^2$ is
\[
  s^2 - 4xz = (x - z)^2 + y^2 + 2y(x + z).
\]
Since $x, y, z \ge 1$, this expression is always strictly positive
($\ge 5$). By Descartes' rule of signs, both roots for $\lambda^2$ are
positive, so \textbf{every} 4-orbit tree possesses exactly 2 positive
eigenvalues.

\subsection{The exact threshold of \texorpdfstring{$T(7,1,9)$}{T(7,1,9)}}

\begin{theorem}[Non-oscillatory threshold]
  \label{thm:threshold}
  Let $H = T(7,1,9)$. Then $\Hom(E_n, H) < \Hom(P_n, H)$ for all
  odd $n \ge 13$, and $\Hom(P_n, H) \le \Hom(E_n, H)$ for all
  $n < 13$ and all even $n$.
\end{theorem}

\begin{proof}
  The characteristic polynomial of the quotient matrix
  \[
    M = \begin{pmatrix} 0&7&0&0 \\ 1&0&1&0 \\
    0&1&0&9 \\ 0&0&1&0 \end{pmatrix}
  \]
  is $t^4 - 17t^2 + 63 = 0$. Setting $u = t^2$:
  \[
    u = \frac{17 \pm \sqrt{37}}{2},
    \qquad
    \lambda_1 = \sqrt{\frac{17 + \sqrt{37}}{2}} \approx 3.3973,
    \qquad
    \lambda_2 = \sqrt{\frac{17 - \sqrt{37}}{2}} \approx 2.3364.
  \]
  For odd $n$, the spectral decomposition yields
  \[
    \Delta_{\mathrm{odd}}(n) \;=\; \Hom(P_n, H) - \Hom(E_n, H)
    \;=\; A\,\lambda_1^n - |B|\,\lambda_2^n,
  \]
  with $A > 0$, $|B| > 0$. Setting
  $r = \lambda_2 / \lambda_1 = \sqrt{(17 - \sqrt{37})/(17 + \sqrt{37})}
  \approx 0.6877$:
  \[
    \Delta_{\mathrm{odd}}(n) = \lambda_1^n \bigl(A - |B|\,r^n\bigr).
  \]
  Since $r^n$ is strictly monotonically decreasing, $A - |B|\,r^n$ can
  cross zero at most once. Evaluating directly:
  \begin{align*}
    \Delta(11) &= 13{,}993{,}266 - 14{,}000{,}700 = -7{,}434 < 0, \\
    \Delta(13) &= 161{,}209{,}734 - 161{,}161{,}476 = +48{,}258 > 0.
  \end{align*}
  Thus $n = 13$ is the unique, permanent threshold.

  For even $n$, the path minimizes due to its perfectly balanced
  bipartition (see Remark~\ref{rem:parity}).
\end{proof}

\begin{remark}[Parity]
  \label{rem:parity}
  For even $n$, $P_n$ has a perfectly balanced bipartition $(n/2, n/2)$,
  minimising its dominant projection. For even depths $d$ (e.g., $d=2$),
  $E_n^{(d)}$ is forced into an unbalanced $(n/2+1, n/2-1)$ bipartition,
  giving it a strictly larger leading coefficient. Even for odd depths
  (e.g., $d=3$) where $E_n^{(d)}$ remains balanced, the path's linear
  geometry yields a smaller projection onto the dominant eigenvector.
  This structural advantage shields the path from being beaten at any
  even~$n$. For odd $n$, both $P_n$ and $E_n^{(d)}$ share the same
  unbalanced bipartition $(\lceil n/2 \rceil, \lfloor n/2 \rfloor)$,
  enabling the secondary eigenvector to interfere and drive the crossover.
\end{remark}

\subsection{The \texorpdfstring{$n = 5$}{n=5} impossibility}

\begin{proposition}[$n = 5$ threshold is algebraically impossible]
  \label{prop:n5}
  For all $T(x,y,z)$ with $x, y, z \ge 1$,
  \[
    \Hom(P_5, T) - \Hom(E_5, T) = -xy\bigl((x - z - 1)^2 + y^2 z\bigr) < 0.
  \]
  In particular, $E_5$ always has \emph{more} homomorphisms than $P_5$
  into any $T(x,y,z)$, so no Leontovich crossover can occur at $n = 5$.
\end{proposition}

\begin{proof}
  Let $M$ denote the $4 \times 4$ quotient matrix of $T(x,y,z)$
  over its four automorphism orbits (root, children, grandchildren,
  leaves), with orbit-size vector
  $\mathbf{s} = (1,\, x,\, xy,\, xyz)^T$:
  \[
    M = \begin{pmatrix} 0&x&0&0 \\ 1&0&y&0 \\
    0&1&0&z \\ 0&0&1&0 \end{pmatrix}.
  \]
  For any path $P_n$, the homomorphism count decomposes as
  $\Hom(P_n, H) = \mathbf{s}^T M^{n-1} \mathbf{1}$,
  where $\mathbf{1} = (1,1,1,1)^T$. Iterating:
  \begin{align*}
    M\,\mathbf{1} &= (x,\; y+1,\; z+1,\; 1)^T, \\
    M^2\mathbf{1} &= (xy+x,\; x+yz+y,\; y+z+1,\; z+1)^T.
  \end{align*}
  The tree $E_5$ has vertices $\{1,2,3,4,5\}$ with edges
  $(1,2),\,(2,3),\,(3,4),\,(2,5)$, so vertex~2 has degree~3. Rooting
  at vertex~2 and summing over orbit assignments:
  \[
    \Hom(E_5, H) = \sum_{b=0}^{3}
      s_b \cdot (M\mathbf{1})_b^2 \cdot (M^2\mathbf{1})_b,
  \]
  since the three branches from vertex~2 contribute
  one factor of $(M\mathbf{1})_b$ each for the two pendant
  leaves (vertices~1 and~5) and one factor of
  $(M^2\mathbf{1})_b$ for the length-2 branch (vertices 3--4).

  Expanding $\mathbf{s}^T M^4 \mathbf{1}$ and collecting terms:
  \begin{align*}
    \Hom(P_5, T) - \Hom(E_5, T)
    &= -xy\bigl(x^2 - 2x(z+1) + (z+1)^2 + y^2 z\bigr) \\
    &= -xy\bigl((x - z - 1)^2 + y^2 z\bigr).
  \end{align*}
  Since $x, y, z \ge 1$, the squared term
  $(x - z - 1)^2 \ge 0$ and $y^2 z \ge 1$, so the
  parenthesised sum is strictly positive. Multiplied by $-xy < 0$,
  the difference is strictly negative.
\end{proof}

\section{Bounding the Smallest Leontovich Graph}
\label{sec:bounds}

\subsection{Exhaustive sweep}

We tested all 1,018,690,325 connected graphs on $m \le 11$
vertices~\cite{oeis_A001349} (generated by McKay's
\texttt{geng}~\cite{mckay2014practical}) against the
$E_n^{(d)}$ filter for $n \le 200$, $d \le 20$.

\begin{theorem}
  No connected simple graph on $m \le 11$ vertices is Leontovich.
\end{theorem}

This computationally resolves Problem~4.3 of Galvin et al.~\cite{galvin2025}
for simple graphs, establishing $m \ge 12$ as the strict lower bound for
the order of any simple Leontovich graph. Because our search space comprised
simple graphs (generated by \texttt{geng}), this bound does not exclude the
possibility of a looped graph on fewer than 12 vertices being strongly
Leontovich in the sense of~\cite{galvin2026}.

\subsection{Asymptotic defense via Perron--Frobenius}

\begin{proposition}
  \label{prop:pf}
  Testing only the near-path family $E_n^{(d)}$ is asymptotically
  sufficient. Highly branched trees cannot beat the path.
\end{proposition}

\begin{proof}
  The path $P_n$ achieves the minimal exponential growth rate
  $\lambda_1(H)^n$ (Csikv\'ari and Lin~\cite{csikvari2015}). For any
  connected, non-regular graph $H$, the Perron--Frobenius eigenvector equation guarantees
  $\Delta(H) > \lambda_1(H)$ strictly. Trees with macroscopic branching
  (such as stars) grow at an exponential rate strictly greater than $\lambda_1(H)^n$,
  exponentially faster than the path.
  To challenge the path, a tree must match the growth base $\lambda_1^n$
  while achieving a smaller leading coefficient---only possible for
  near-path trees $E_n^{(d)}$ with localized boundary defects.
\end{proof}

This reduces the per-target cost from brute-force $O(t \cdot n \cdot m^2)$
(over $t$ source trees) to $O(d_{\max} \cdot n_{\max} \cdot m)$ via
$O(m)$ matrix--vector iteration up to $n = 200$, $d \le 20$.
Filtering across all 1,018,690,325 connected graphs with $m \le 11$
yields zero violations.

\section{Computational Methodology}
\label{sec:computation}

Tree enumeration uses the Beyer--Hedetniemi canonical level-sequence
algorithm, generating all non-isomorphic trees on $n$ vertices in
$O(n)$ amortized time per tree. The \texttt{synthesizer} engine outputs
trees in graph6 format on \texttt{stdout}, which is piped to the
\texttt{leontovich\_fast} filter. The filter decodes graph6 strings,
precomputes $A^k \mathbf{1}$ for $k = 0, \ldots, n_{\max}$, and evaluates
$\Hom(E_n^{(d)}, H)$ vs.\ $\Hom(P_n, H)$ via matrix--vector iteration
in IEEE 754 double precision.
Comparisons use a relative tolerance of $10^{-11}$, which exceeds the
worst-case accumulated rounding error ($\sim n_{\max} \cdot m \cdot
\varepsilon \approx 6.7 \times 10^{-13}$ at $m = 15$, $n = 200$) by
a factor of~15.  All flagged anomalies are subsequently verified with
exact integer arithmetic.
The total cost per target graph is $O(n_{\max} \cdot m^2)$, giving
$O(h \cdot n_{\max} \cdot m^2)$ for the full sweep over $h$ targets.

Throughput: $\sim\!256{,}000$ graphs/second on six threads (Intel
Coffee Lake, AVX2 FMA) using dense $16 \times 16$ padded adjacency
matrices, achieving near-peak memory bandwidth via optimized DAXPY loop
ordering and OpenMP parallelism. The bipartite sweep
of 608,930,559 graphs on $m \le 15$ vertices (validated against OEIS
A005142~\cite{oeis}) completes in approximately 35~minutes.

\begin{remark}[Depth anomalies]
\label{rem:depth_anomaly}
While no $d = 2$ Leontovich violations were found, the $m = 15$
bipartite sweep revealed a single graph $H^*$ (Figure~\ref{fig:anomaly})
for which $\Hom(E_n^{(d)}, H^*) < \Hom(P_n, H^*)$ at $d \ge 14$,
verified with exact integer arithmetic.
This bipartite graph on 15 vertices with partition sizes $(3,12)$
and degree sequence $[7, 5, 5, 2^5, 1^7]$ exhibits 300
such anomalies across $d \in \{14,\ldots,20\}$ and odd
$n \in \{57,\ldots,199\}$, each with relative deficit
$\sim 7.6 \times 10^{-4}$.
No anomalies of any depth were detected at $m \le 13$.
\end{remark}

\end{remark}

\subsection*{Symmetry breaking and the active quotient spectrum}

The spectral structure of the pruning sequence admits a clean analytic
description that resolves the crossover behaviour. In the fully symmetric
graph $T(7,1,[9^7])$, the adjacency matrix possesses highly degenerate
eigenvalues at $\lambda = \sqrt{10}$ (arising from the isolated 9-leaf hubs).
However, the eigenvectors associated with this eigenspace are antisymmetric
across the identical branches. Consequently, they are perfectly orthogonal
to the all-ones vector $\mathbf{1}$ and the initial degree vector, meaning
their projection coefficients in the transfer matrix expansion
$\Hom(G_n, H) = \mathbf{1}^T A^{n-4} \mathbf{h}_0$ are identically zero.
These ``dark'' eigenvalues contribute nothing to the homomorphism counts.

Instead, the crossover dynamics are governed exclusively by the \emph{active}
spectrum---the eigenvalues of the quotient matrix over the graph's
automorphism orbits. For the fully symmetric graph $T(7,1,[9^7])$, the
active secondary eigenvalue is $\lambda_2^* \approx 2.3364$.

When leaves are pruned asymmetrically, the perfect identical-branch symmetry
is broken. For $T(7,1,[9^k, 8^{7-k}])$, the orbits split and the quotient
matrix expands from $4 \times 4$ to $7 \times 7$. This symmetry breaking
activates a new secondary eigenvalue $\lambda_2^*$ that directly interferes
with the Perron mode.

The Leontovich crossover arises from destructive interference between the
dominant Perron mode $\lambda_1^n$ and the secondary active mode $(\lambda_2^*)^n$
in the difference $\Hom(P_n, H) - \Hom(E_n^{(d)}, H)$. As branches are pruned,
$\lambda_1$ decreases while the active secondary eigenvalue $\lambda_2^*$
increases (see Table~\ref{tab:sa_hits}), narrowing the spectral gap.
At $|V| = 76$, the interference from $\lambda_2^* \approx 3.0380$ is just
sufficient to drive the ratio $\Hom(E_n^{(2)}, H)/\Hom(P_n, H)$ below~1 at
$n = 17$. However, at $|V| = 75$, the gap has narrowed too much ($\lambda_2^* = 3.0581$),
suppressing the secondary interference coefficient so the Perron mode
of the path strictly dominates at all odd $n$, and the crossover never occurs.

This yields a sharp structural characterisation: \emph{within the star-of-stars
basin, asymmetric pruning breaks orbital symmetry and lifts new active
eigenvalues into the quotient spectrum.} The Leontovich property is lost when
this symmetry-breaking pushes the active secondary eigenvalue too close to the
spectral radius, establishing the 76-vertex graph as the unique minimal size
in this pruning sequence.

\subsection*{Deeper symmetric trees}

The 76-vertex graph $T(7,1,[9^5,8^2])$ is the smallest known
Leontovich graph, but it is not the only small one outside the
$T(x,1,z)$ family.  The 5-orbit tree $T(1,7,1,10)$, with branching
sequence $[1,7,1,10]$, has $|V| = 86$ and is Leontovich with first
crossover at $n = 15$ ($d = 4$) and active quotient eigenvalue
$\lambda_2^* \approx 2.541$.  This
confirms that deeper symmetric trees can achieve the Leontovich
property at moderate size, though none yet found are smaller than
the 76-vertex asymmetric star-of-stars.

\subsection*{Non-oscillatory threshold}

\begin{proposition}
For a bipartite tree $H$ with exactly two dominant positive
eigenvalues $\lambda_1 > \lambda_2 > 0$, the difference
$\Delta_n = \Hom(P_n, H) - \Hom(E_n^{(d)}, H)$ restricted to
odd~$n$ can change sign at most once.
\end{proposition}

\begin{proof}[Proof sketch]
By spectral decomposition, the homomorphism counts for large $n$
are dominated by two terms:
\[
  \Delta_{\mathrm{odd}}(n) = A\,\lambda_1^n + B\,\lambda_2^n
  = \lambda_1^n \!\left( A - |B|\left(\frac{\lambda_2}{\lambda_1}\right)^{\!n}\right)
\]
where $A > 0$ (since $P_n$ is the asymptotic minimiser) and $B < 0$
(since $E_n^{(d)}$ wins at small~$n$).  Because
$0 < \lambda_2/\lambda_1 < 1$, the subtracted term
$|B|(\lambda_2/\lambda_1)^n$ is strictly decreasing.  Thus the
parenthesised expression crosses zero at most once: once $\Delta$
becomes positive, it remains positive for all larger odd~$n$.
\end{proof}

\noindent
We verified this computationally for the 76-vertex graph: $\Delta$
is negative for $n \le 15$ and strictly positive for all
$n \ge 17$ (checked through $n = 43$), with the gap growing
exponentially from $\Delta_{17} = 1{,}993{,}646$ to
$\Delta_{43} > 10^{17}$.

\subsection*{Universal bipartite parity rule}

\begin{proposition}
For any bipartite target graph $H$, $\Hom(E_n^{(d)}, H) \ge
\Hom(P_n, H)$ for all even~$n$.  That is, the path $P_n$ is
always the minimiser among trees on $n$ vertices when $n$ is even
and $H$ is bipartite.
\end{proposition}

\noindent
When $n$ is even, $P_n$ has a perfectly balanced bipartition $(n/2, n/2)$,
minimising its projection onto $H$'s dominant eigenvector.  For even
depths~$d$ (e.g.\ $d=2$), $E_n^{(d)}$ is forced into an unbalanced
$(n/2+1, n/2-1)$ bipartition, giving it a strictly larger leading
coefficient.  Even for odd depths (e.g.\ $d=3$) where $E_n^{(d)}$
remains balanced, the path's linear geometry uniquely minimises the
projection coefficient.  For odd~$n$, both trees share the same
unbalanced bipartition, allowing the secondary eigenvector to interfere
and flip the overall count.

We verified this computationally: across all target graphs tested
(over 463~million bipartite graphs and 5.6~million trees), no
even-$n$ crossover was observed.  This is consistent with the
observation of Galvin et al.~\cite{galvin2026} that adding a loop to the root
of $H$ (destroying bipartiteness) is necessary to achieve even-$n$ crossovers.

\begin{remark}
The same parity phenomenon appears in the independent set
setting: the consecutive ratios $a(N)/a(N-1)$ of maximum independent
set counts on degree-constrained trees oscillate with period~2,
reflecting the same bipartite balance constraint.  Since an
independent set is a homomorphism into the non-bipartite looped graph $H_{\mathrm{ind}}$
(which breaks bipartiteness via its loop), this explains why even-$n$ crossovers
are possible in that setting.
\end{remark}
